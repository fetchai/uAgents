77async def register_in_agentverse( 

78 request: AgentverseConnectRequest, 

79 identity: Identity, 

80 prefix: AddressPrefix, 

81 agentverse: AgentverseConfig, 

82 agent_details: AgentUpdates | None = None, 

83) -> RegistrationResponse: 

84 """ 

85 Registers agent in Agentverse 

86 

87 Args: 

88 request (AgentverseConnectRequest): Request object 

89 identity (Identity): Agent identity object 

90 prefix (AddressPrefix): Agent address prefix 

91 can be "agent" (mainnet) or "test-agent" (testnet) 

92 agentverse (AgentverseConfig): Agentverse configuration 

93 agent_details (AgentUpdates | None): Agent details (name, readme, avatar_url) 

94 

95 Returns: 

96 RegistrationResponse: Registration response object 

97 """ 

98 async with aiohttp.ClientSession() as session: 

99 # get challenge 

100 challenge_url = f"{agentverse.url}/v1/auth/challenge" 

101 challenge_request = ChallengeRequest(address=identity.address) 

102 logger.debug("Requesting mailbox access challenge") 

103 async with session.post( 

104 challenge_url, 

105 data=challenge_request.model_dump_json(), 

106 headers={ 

107 "content-type": "application/json", 

108 "Authorization": f"Bearer {request.user_token}", 

109 }, 

110 ) as resp: 

111 resp.raise_for_status() 

112 challenge = ChallengeResponse.model_validate_json(await resp.text()) 

113 

114 # response to challenge with signature to get token 

115 prove_url = f"{agentverse.url}/v1/agents" 

116 async with session.post( 

117 url=prove_url, 

118 data=RegistrationRequest( 

119 address=identity.address, 

120 prefix=prefix, 

121 challenge=challenge.challenge, 

122 challenge_response=identity.sign(challenge.challenge.encode()), 

123 endpoint=request.endpoint, 

124 agent_type=request.agent_type, 

125 ).model_dump_json(), 

126 headers={ 

127 "content-type": "application/json", 

128 "Authorization": f"Bearer {request.user_token}", 

129 }, 

130 ) as resp: 

131 if resp.status == 200: 

132 logger.info( 

133 f"Successfully registered as {request.agent_type} agent in Agentverse" 

134 ) 

135 if agent_details: 

136 await update_agent_details( 

137 request.user_token, identity.address, agent_details, agentverse 

138 ) 

139 return RegistrationResponse(success=True) 

140 

141 detail = (await resp.json())["detail"] 

142 return RegistrationResponse(success=False, detail=detail)

212class MailboxClient: 

213 """Client for interacting with the Agentverse mailbox server."""

215 def __init__( 

216 self, 

217 identity: Identity, 

218 agentverse: AgentverseConfig, 

219 logger: logging.Logger | None = None, 

220 ): 

221 self._identity = identity 

222 self._agentverse = agentverse 

223 self._access_token: str | None = None 

224 self._poll_interval = MAILBOX_POLL_INTERVAL_SECONDS 

225 self._logger = logger or get_logger("mailbox")

227 async def run(self): 

228 """Runs the mailbox client.""" 

229 self._logger.info(f"Starting mailbox client for {self._agentverse.url}") 

230 loop = asyncio.get_event_loop() 

231 loop.create_task(self._check_mailbox_loop())

233 async def _check_mailbox_loop(self): 

234 """Retrieves envelopes from the mailbox server and processes them.""" 

235 while True: 

236 try: 

237 async with aiohttp.ClientSession() as session: 

238 if self._access_token is None: 

239 await self._get_access_token() 

240 mailbox_url = f"{self._agentverse.url}/v1/mailbox" 

241 async with session.get( 

242 mailbox_url, 

243 headers={ 

244 "Authorization": f"token {self._access_token}", 

245 }, 

246 ) as resp: 

247 success = resp.status == 200 

248 if success: 

249 items = (await resp.json())["items"] 

250 for item in items: 

251 stored_env = StoredEnvelope.model_validate(item) 

252 await self._handle_envelope(stored_env) 

253 elif resp.status == 401: 

254 self._access_token = None 

255 self._logger.warning( 

256 "Access token expired: a new one should be retrieved automatically" 

257 ) 

258 else: 

259 self._logger.debug( 

260 f"Failed to retrieve messages: {resp.status}:{(await resp.text())}" 

261 ) 

262 except ClientConnectorError as ex: 

263 self._logger.warning(f"Failed to connect to mailbox server: {ex}") 

264 

265 await asyncio.sleep(self._poll_interval)

267 async def _handle_envelope(self, stored_env: StoredEnvelope): 

268 """ 

269 Handles an envelope received from the mailbox server. 

270 Dispatches the incoming messages and adds the envelope to the deletion queue. 

271 

272 Args: 

273 stored_env (StoredEnvelope): Envelope to handle 

274 """ 

275 try: 

276 env = Envelope.model_validate(stored_env.envelope) 

277 except ValidationError: 

278 self._logger.warning("Received invalid envelope") 

279 return 

280 

281 if not is_user_address(env.sender): # verify signature if sent from agent 

282 try: 

283 env.verify() 

284 except Exception as err: 

285 self._logger.warning( 

286 "Received envelope that failed verification: %s", err 

287 ) 

288 return 

289 

290 if not dispatcher.contains(env.target): 

291 self._logger.warning("Received envelope for unrecognized address") 

292 return 

293 

294 await dispatcher.dispatch_msg( 

295 sender=env.sender, 

296 destination=env.target, 

297 schema_digest=env.schema_digest, 

298 message=env.decode_payload(), 

299 session=env.session, 

300 ) 

301 

302 # delete envelope from server 

303 await self._delete_envelope(stored_env.uuid)

325 ) 

326 except ClientConnectorError as ex:

333 async with aiohttp.ClientSession() as session: 

334 challenge_url = f"{self._agentverse.url}/v1/auth/challenge"

361 if resp and resp.status == 200: 

362 challenge_proof_response = ChallengeProofResponse.parse_raw( 

363 await resp.text()

365 self._logger.info("Mailbox access token acquired")

367 else:
